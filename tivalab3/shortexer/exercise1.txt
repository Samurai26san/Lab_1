#include <stdint.h>
#include <stdbool.h>
#include "inc/hw_memmap.h"
#include "driverlib/sysctl.h"
#include "driverlib/gpio.h"

#define LED1 GPIO_PIN_0   // PN0
#define LED2 GPIO_PIN_1   // PN1

int main(void) {
    uint32_t sysclk;

    sysclk = SysCtlClockFreqSet(SYSCTL_XTAL_25MHZ | SYSCTL_OSC_MAIN |
                                SYSCTL_USE_PLL | SYSCTL_CFG_VCO_320, 120000000);

    SysCtlPeripheralEnable(SYSCTL_PERIPH_GPION);
    while(!SysCtlPeripheralReady(SYSCTL_PERIPH_GPION));

    GPIOPinTypeGPIOOutput(GPIO_PORTN_BASE, LED1 | LED2);

    while(1) {
        GPIOPinWrite(GPIO_PORTN_BASE, LED1, LED1);  // Encender PN0
        SysCtlDelay(sysclk/3);                      // ~1 segundo

        GPIOPinWrite(GPIO_PORTN_BASE, LED2, LED2);  // Encender PN1
        SysCtlDelay(sysclk/3);                      // ~1 segundo

        GPIOPinWrite(GPIO_PORTN_BASE, LED1|LED2, 0); // Apagar ambos
        SysCtlDelay(sysclk/3);                       // ~1 segundo
    }
}
